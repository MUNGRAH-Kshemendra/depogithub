pipeline {
    agent any

    stages {

        stage('Clean workspace') {
            steps {
                cleanWs()
            }
        }

        stage('Checkout SCM') {
            steps {
                git branch: 'main',
                    credentialsId: 'kshemendra-github-token',
                    url: 'https://github.com/MUNGRAH-Kshemendra/depogithub.git'
            }
        }

        stage('Reset Docker (optional)') {
            steps {
                sh '''
                sudo apt remove -y docker.io docker-compose
                sudo apt autoremove -y
                '''
            }
        }

        stage('Install Docker') {
            steps {
                sh '''
                sudo apt update
                sudo apt install -y docker.io
                sudo systemctl start docker
                '''
            }
        }

        stage('Run Container') {
            steps {
                sh '''
                docker run -d --name conteneur01 -p 8080:80 httpd
                docker ps
                '''
            }
        }
    }
}
