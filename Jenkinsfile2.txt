pipeline {
    agent any

    stages {

        stage('Clean workspace') {
            steps {
                cleanWs()
            }
        }

        stage('Checkout SCM') {
            steps {
                git branch: 'main',
                    credentialsId: 'kshemendra-github-token',
                    url: 'https://github.com/MUNGRAH-Kshemendra/depogithub.git'
            }
        }

        stage('installer docker') {
            steps {
                sh 'sudo apt autoremove docker.io docker-compose -y'
                sh 'sudo apt update'
                sh 'sudo apt upgrade -y'
                sh 'sudo apt install apt-transport-https ca-certificates curl software-properties-common -y'
                sh 'sudo apt install docker.io -y'
                sh 'sudo apt update'
                sh 'sudo usermod -aG docker jenkins'
                sh 'sudo systemctl status docker'
                sh 'sudo systemctl start docker'
                sh 'sudo systemctl restart docker'
            }
        }

        stage('Remove old container if exists') {
            steps {
                sh 'docker stop conteneur01 || true'
                sh 'docker rm conteneur01 || true'
            }
        }

        stage('Run Container') {
            steps {
                sh 'docker ps'
                sh 'docker pull ubuntu:latest'
                sh 'sudo docker run -d --name conteneur01 --hostname conteneur01 ubuntu:latest'
            }
        }
    }
}
